import pandas as pd import matplotlib.pyplot as plt import numpy as np # 1. 準備資料 (您可以替換成您的 Excel 或 CSV 檔案) data = { '姓名': [f'學生{i}' for i in range(1, 51)], '成績': np.random.normal(75, 12, 50).clip(40, 100).round(1) } df = pd.DataFrame(data) # 2. 計算描述性統計 stats = df['成績'].describe() print(stats) # 3. 定義分數區間並計算分佈 bins = [0, 60, 70, 80, 90, 100] labels = ['不及格 (<60)', '60-69', '70-79', '80-89', '優異 (90-100)'] df['分佈區間'] = pd.cut(df['成績'], bins=bins, labels=labels, right=False) dist = df['分佈區間'].value_counts().sort_index() # 4. 繪製長條圖 plt.figure(figsize=(10, 6)) dist.plot(kind='bar', color='skyblue', edgecolor='black') plt.title('班級成績分佈圖', fontsize=15) plt.ylabel('人數') plt.tight_layout() plt.savefig('score_distribution.png') # 5. 繪製盒鬚圖 plt.figure(figsize=(6, 8)) plt.boxplot(df['成績'], patch_artist=True) plt.title('成績盒鬚圖 (Boxplot)', fontsize=15) plt.ylabel('分數') plt.savefig('score_boxplot.png') # 6. 匯出分析結果 df.to_csv('student_grades_analysis.csv', index=False)
